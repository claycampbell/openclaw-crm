3. **Create dynamic Zod schema builder** at `apps/web/src/lib/record-validation.ts`:
   ```typescript
   import { z } from "zod";

   interface AttributeDef {
     id: string;
     slug: string;
     title: string;
     type: string;
     isRequired: boolean;
     isMultiselect: boolean;
     options?: { id: string; title: string; color: string }[];
     statuses?: { id: string; title: string; color: string; isActive: boolean }[];
   }

   /**
    * Build a Zod schema from EAV attribute definitions.
    * Each attribute becomes a field in the schema with appropriate validation.
    */
   export function buildRecordSchema(attributes: AttributeDef[]): z.ZodObject<Record<string, z.ZodTypeAny>> {
     const shape: Record<string, z.ZodTypeAny> = {};

     for (const attr of attributes) {
       let field: z.ZodTypeAny;

       switch (attr.type) {
         case "email":
           field = z.string().email(`${attr.title} must be a valid email`);
           break;
         case "url":
           field = z.string().url(`${attr.title} must be a valid URL`);
           break;
         case "number":
         case "currency":
         case "rating":
           field = z.coerce.number({ invalid_type_error: `${attr.title} must be a number` });
           break;
         case "date":
         case "timestamp":
           field = z.string().min(1, `${attr.title} is required`);
           break;
         case "boolean":
           field = z.boolean();
           break;
         case "phone":
           field = z.string().min(1, `${attr.title} is required`);
           break;
         case "select":
         case "status":
           field = z.string();
           break;
         case "multiselect":
           field = z.array(z.string());
           break;
         case "json":
           field = z.any();
           break;
         case "record_reference":
         case "actor":
           field = z.string();
           break;
         default: // text, textarea
           field = z.string();
           break;
       }

       // Apply required/optional
       if (attr.isRequired) {
         if (field instanceof z.ZodString) {
           field = field.min(1, `${attr.title} is required`);
         }
       } else {
         field = field.optional().or(z.literal("")).or(z.null());
       }

       shape[attr.slug] = field;
     }

     return z.object(shape);
   }
   ```

   Handle edge cases:
   - Optional fields should accept empty string, null, or undefined
   - Number fields use z.coerce.number() so string inputs are coerced
   - Email validation only for non-empty optional email fields
   - Multiselect fields accept arrays
