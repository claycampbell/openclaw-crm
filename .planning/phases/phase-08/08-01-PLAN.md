Replace the placeholder `ai_generate` handler in `instrumentation.ts` with a real dispatcher that routes by `documentType`:

```typescript
registerJobHandler("ai_generate", async (payload) => {
  const { workspaceId, recordId, documentType, meetingId, triggerType, noteText, competitorName } = payload as AiGeneratePayload & { workspaceId: string };
  if (!workspaceId || !recordId) throw new Error("ai_generate missing workspaceId or recordId");

  switch (documentType) {
    case "opportunity_brief":
      await generateOpportunityBrief(workspaceId, recordId);
      break;
    case "proposal":
      await generateProposal(workspaceId, recordId);
      break;
    case "deck":
      await generateDeck(workspaceId, recordId);
      break;
    case "meeting_prep":
      await generateMeetingPrepBrief(workspaceId, recordId, meetingId!);
      break;
    case "followup":
      await generatePostMeetingFollowup(workspaceId, recordId, {
        type: (triggerType as "meeting_ended" | "note_added") ?? "meeting_ended",
        noteText,
      });
      break;
    case "battlecard":
      await generateBattlecard(workspaceId, recordId, competitorName!);
      break;
    default:
      console.warn(`[ai_generate] Unknown documentType: ${documentType}`);
  }
});
```

Create `apps/web/src/services/documents/meeting-prep-cron.ts`:

This service finds calendar events starting in the next 25-35 minutes (a 10-minute window centered on T-30min) where `prepJobEnqueued` is false, and enqueues an `ai_generate` job with `documentType: "meeting_prep"` for each. After enqueuing, it sets `prepJobEnqueued = true` on the calendar event to prevent re-enqueuing.

```typescript
import { db } from "@/db";
import { calendarEvents } from "@/db/schema";
import { and, eq, gte, lte, isNotNull } from "drizzle-orm";
import { enqueueJob } from "@/lib/job-queue";

export async function processMeetingPrepJobs(): Promise<number> {
  const now = new Date();
  const windowStart = new Date(now.getTime() + 25 * 60 * 1000); // 25 min from now
  const windowEnd = new Date(now.getTime() + 35 * 60 * 1000);   // 35 min from now

  // Find meetings starting in 25-35 minutes that haven't had prep enqueued
  const upcomingMeetings = await db
    .select()
    .from(calendarEvents)
    .where(
      and(
        gte(calendarEvents.startAt, windowStart),
        lte(calendarEvents.startAt, windowEnd),
        eq(calendarEvents.prepJobEnqueued, false),
        isNotNull(calendarEvents.recordId) // Must be linked to a deal
      )
    )
    .limit(20);

  let enqueued = 0;
  for (const meeting of upcomingMeetings) {
    if (!meeting.recordId || !meeting.workspaceId) continue;

    await enqueueJob(meeting.workspaceId, "ai_generate", {
      documentType: "meeting_prep",
      recordId: meeting.recordId,
      contextTier: "light",
      meetingId: meeting.id,
    });

    // Mark as enqueued to prevent duplicates
    await db
      .update(calendarEvents)
      .set({ prepJobEnqueued: true })
      .where(eq(calendarEvents.id, meeting.id));

    enqueued++;
  }

  return enqueued;
}
```

Create `apps/web/src/app/api/v1/cron/meeting-prep/route.ts`:

Standard cron endpoint pattern (same as `/api/v1/cron/jobs`):

```typescript
import { NextRequest, NextResponse } from "next/server";
import { processMeetingPrepJobs } from "@/services/documents/meeting-prep-cron";

export async function GET(req: NextRequest) {
  const cronSecret = process.env.CRON_SECRET;
  if (cronSecret) {
    const authHeader = req.headers.get("authorization");
    if (authHeader !== `Bearer ${cronSecret}`) {
      return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
    }
  }

  const enqueued = await processMeetingPrepJobs();
  return NextResponse.json({
    enqueued,
    timestamp: new Date().toISOString(),
  });
}
```

This cron endpoint should be called every 5 minutes (via external cron service or Vercel cron). The 10-minute window (25-35 min) ensures a meeting is caught even if the cron runs a few minutes early or late.
