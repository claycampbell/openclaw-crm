    1. Create apps/web/src/services/job-queue.ts:

    ```typescript
    import { db } from "@/db";
    import { backgroundJobs } from "@/db/schema";
    import { eq, and, lte, sql } from "drizzle-orm";
    import { addMinutes } from "date-fns";

    export type JobType = "signal_evaluate" | "ai_generate" | "email_send" | "email_sync" | "calendar_sync" | "transcript_ingest";

    export interface JobPayload {
      workspaceId: string;
      [key: string]: unknown;
    }

    export interface EnqueueOptions {
      idempotencyKey?: string;
      runAt?: Date;
      maxRetries?: number;
    }

    // Handler registry — job type maps to async handler function
    type JobHandler = (payload: Record<string, unknown>) => Promise<void>;
    const handlers = new Map<JobType, JobHandler>();

    export function registerJobHandler(type: JobType, handler: JobHandler): void {
      handlers.set(type, handler);
    }

    export async function enqueueJob(
      type: JobType,
      payload: JobPayload,
      options: EnqueueOptions = {}
    ): Promise<string> {
      const { idempotencyKey, runAt = new Date(), maxRetries = 3 } = options;

      // Deduplication: if idempotency key provided and job exists, return existing id
      if (idempotencyKey) {
        const existing = await db
          .select({ id: backgroundJobs.id })
          .from(backgroundJobs)
          .where(eq(backgroundJobs.idempotencyKey, idempotencyKey))
          .limit(1);
        if (existing.length > 0) return existing[0].id;
      }

      const [job] = await db
        .insert(backgroundJobs)
        .values({
          workspaceId: payload.workspaceId,
          type,
          payload: payload as Record<string, unknown>,
          status: "pending",
          runAt,
          maxRetries,
          idempotencyKey: idempotencyKey ?? null,
        })
        .returning({ id: backgroundJobs.id });

      return job.id;
    }

    export async function processJobs(batchSize = 10): Promise<number> {
      // Claim pending jobs that are due — using FOR UPDATE SKIP LOCKED to prevent double-processing
      const claimed = await db.execute(sql`
        UPDATE background_jobs
        SET status = 'running', started_at = NOW()
        WHERE id IN (
          SELECT id FROM background_jobs
          WHERE status = 'pending' AND run_at <= NOW()
          ORDER BY run_at ASC
          LIMIT ${batchSize}
          FOR UPDATE SKIP LOCKED
        )
        RETURNING *
      `);

      const jobs = claimed.rows as Array<typeof backgroundJobs.$inferSelect>;
      let processed = 0;

      for (const job of jobs) {
        const handler = handlers.get(job.type as JobType);
        if (!handler) {
          // No handler registered — mark failed immediately
          await db
            .update(backgroundJobs)
            .set({ status: "failed", error: `No handler registered for job type: ${job.type}`, completedAt: new Date() })
            .where(eq(backgroundJobs.id, job.id));
          continue;
        }

        try {
          await handler(job.payload as Record<string, unknown>);
          await db
            .update(backgroundJobs)
            .set({ status: "completed", completedAt: new Date() })
            .where(eq(backgroundJobs.id, job.id));
          processed++;
        } catch (err) {
          const nextRetryCount = (job.retryCount ?? 0) + 1;
          const maxRetries = job.maxRetries ?? 3;
          const isExhausted = nextRetryCount >= maxRetries;
          // Exponential backoff: 5 * 2^retryCount minutes (5, 10, 20 min)
          const backoffMinutes = 5 * Math.pow(2, job.retryCount ?? 0);
          await db
            .update(backgroundJobs)
            .set({
              status: isExhausted ? "failed" : "pending",
              retryCount: nextRetryCount,
              error: err instanceof Error ? err.message : String(err),
              runAt: isExhausted ? undefined : addMinutes(new Date(), backoffMinutes),
              completedAt: isExhausted ? new Date() : undefined,
            })
            .where(eq(backgroundJobs.id, job.id));
        }
      }

      return processed;
    }
    ```

    import { NextRequest, NextResponse } from "next/server";
    import { processJobs } from "@/services/job-queue";

    export const runtime = "nodejs";  // Required: FOR UPDATE SKIP LOCKED needs a persistent connection

    export async function GET(req: NextRequest) {
      // Validate cron secret to prevent unauthorized triggers
      const cronSecret = process.env.CRON_SECRET;
      if (cronSecret) {
        const authHeader = req.headers.get("authorization");
        if (authHeader !== `Bearer ${cronSecret}`) {
          return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
        }
      }

      const processed = await processJobs(10);
      return NextResponse.json({ processed, timestamp: new Date().toISOString() });
    }
