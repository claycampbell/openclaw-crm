      const rules = await db
        .select()
        .from(automationRules)
        .where(eq(automationRules.workspaceId, ctx.workspaceId))

      return success(rules);

    export async function POST(req: NextRequest) {
      const ctx = await getAuthContext(req);
      if (!ctx) return unauthorized();
      const adminCheck = requireAdmin(ctx);
      if (adminCheck) return adminCheck;

      const body = await req.json();
      const { name, triggerType, conditions, actionType, actionPayload } = body;

      if (!name || !triggerType || !actionType) {
        return badRequest("name, triggerType, and actionType are required");
      }

      const [rule] = await db
        .insert(automationRules)
        .values({
          workspaceId: ctx.workspaceId,
          name,
          triggerType,
          conditions: conditions ?? [],
          actionType,
          actionPayload: actionPayload ?? {},
          createdBy: ctx.userId,
        })
        .returning();

      return success(rule, 201);
    }

    1. Create apps/web/src/services/automation-engine.ts:

    ```typescript
    import { db } from "@/db";
    import { signalEvents, automationRules } from "@/db/schema";
    import { eq, and, isNull } from "drizzle-orm";
    import { enqueueJob, JobType } from "./job-queue";

    interface RuleCondition {
      field: string;     // dot-path into signal payload, e.g. "to" or "objectSlug"
      operator: "equals" | "not_equals" | "contains" | "exists";
      value?: unknown;
    }

    /**
     * Evaluate conditions against signal payload.
     * All conditions must match (AND semantics).
     * Empty conditions array = always match.
     */
    function matchesConditions(
      conditions: RuleCondition[],
      payload: Record<string, unknown>
    ): boolean {
      if (!conditions || conditions.length === 0) return true;

      return conditions.every((condition) => {
        const fieldValue = getNestedValue(payload, condition.field);
        switch (condition.operator) {
          case "equals":
            return fieldValue === condition.value;
          case "not_equals":
            return fieldValue !== condition.value;
          case "contains":
            return typeof fieldValue === "string" &&
              typeof condition.value === "string" &&
              fieldValue.includes(condition.value);
          case "exists":
            return fieldValue !== undefined && fieldValue !== null;
          default:
            return false;
        }
      });
    }

    function getNestedValue(obj: Record<string, unknown>, path: string): unknown {
      return path.split(".").reduce<unknown>((current, key) => {
        if (current && typeof current === "object") {
          return (current as Record<string, unknown>)[key];
        }
        return undefined;
      }, obj);
    }

    /**
     * Given a signal event ID, evaluate matching automation rules and dispatch jobs.
     * Marks the signal as processed when done.
     * Called by the signal_evaluate job handler.
     */
    export async function evaluateSignalById(signalEventId: string): Promise<void> {
      // Load the signal event
      const [signal] = await db
        .select()
        .from(signalEvents)
        .where(eq(signalEvents.id, signalEventId))
        .limit(1);

      if (!signal) {
        console.warn(`[automation-engine] Signal not found: ${signalEventId}`);
        return;
      }

      // Load enabled rules for this workspace and trigger type
      const rules = await db
        .select()
        .from(automationRules)
        .where(
          and(
            eq(automationRules.workspaceId, signal.workspaceId),
            eq(automationRules.triggerType, signal.type),
            eq(automationRules.enabled, true)
          )
        );

      // Evaluate each rule's conditions against the signal payload
      for (const rule of rules) {
        const conditions = (rule.conditions as RuleCondition[]) ?? [];
        const payload = (signal.payload as Record<string, unknown>) ?? {};

        if (!matchesConditions(conditions, payload)) continue;

        // Map action type to job type
        const jobTypeMap: Record<string, JobType> = {
          enqueue_ai_generate: "ai_generate",
          enqueue_email_send: "email_send",
          enqueue_email_sync: "email_sync",
          enqueue_calendar_sync: "calendar_sync",
        };

        const jobType = jobTypeMap[rule.actionType];
        if (!jobType) {
          // create_task and create_note handled inline (future plan)
          console.log(`[automation-engine] Non-job action ${rule.actionType} for rule ${rule.id} — not yet implemented`);
          continue;
        }

        await enqueueJob(
          jobType,
          {
            workspaceId: signal.workspaceId,
            recordId: signal.recordId,
            signalEventId: signal.id,
            ruleId: rule.id,
            ...(rule.actionPayload as Record<string, unknown>),
          },
          {
            idempotencyKey: `${jobType}:${signal.id}:${rule.id}`,
          }
        );
      }

      // Mark signal as processed (regardless of whether rules matched)
      await db
        .update(signalEvents)
        .set({ processedAt: new Date() })
        .where(eq(signalEvents.id, signalEventId));
    }
    ```
