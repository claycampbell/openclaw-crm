    export const signalTypeEnum = pgEnum("signal_type", [
      "record_created",
      "stage_changed",
      "note_added",
      "email_received",
      "email_sent",
      "email_opened",
      "email_link_clicked",
      "meeting_held",
      "call_completed",
      "task_completed",
      "enrichment_completed",
    ]);

    export const signalEvents = pgTable(
      "signal_events",
      {
        id: text("id").primaryKey().$defaultFn(() => crypto.randomUUID()),
        workspaceId: text("workspace_id").notNull(),
        type: signalTypeEnum("type").notNull(),
        recordId: text("record_id"),           // nullable — some signals are workspace-level
        payload: jsonb("payload").notNull().default({}),
        processedAt: timestamp("processed_at"), // null = not yet consumed by automation engine
        createdAt: timestamp("created_at").notNull().defaultNow(),
      },
      (table) => [
        index("signal_events_workspace_type").on(table.workspaceId, table.type),
        index("signal_events_record_id").on(table.recordId),
        index("signal_events_processed_at").on(table.processedAt),
        index("signal_events_created_at").on(table.createdAt),
      ]
    );

    // Deduplication table for external signals (webhook at-least-once delivery)
    export const processedSignals = pgTable(
      "processed_signals",
      {

    1. Create apps/web/src/services/signals.ts:

    ```typescript
    import { db } from "@/db";
    import { signalEvents, processedSignals } from "@/db/schema";
    import { eq, and } from "drizzle-orm";
    import { enqueueJob } from "./job-queue";

    export type SignalType = typeof signalEvents.$inferInsert["type"];

    export interface WriteSignalEventParams {
      workspaceId: string;
      type: SignalType;
      recordId?: string;
      payload?: Record<string, unknown>;
    }

    /**
     * Write a signal event and enqueue a signal_evaluate job.
     * Call this transactionally alongside any CRM state mutation.
     * Safe to call outside a transaction when the mutation has already committed.
     */
    export async function writeSignalEvent(params: WriteSignalEventParams): Promise<string> {
      const { workspaceId, type, recordId, payload = {} } = params;

      const [event] = await db
        .insert(signalEvents)
        .values({
          workspaceId,
          type,
          recordId: recordId ?? null,
          payload,
        })
        .returning({ id: signalEvents.id });

      // Enqueue automation evaluation job — automation engine (plan 01-03) will handle it
      await enqueueJob(
        "signal_evaluate",
        { workspaceId, signalEventId: event.id },
        { idempotencyKey: `signal_evaluate:${event.id}` }
      );

      return event.id;
    }

    /**
     * Deduplicate an external signal using ON CONFLICT DO NOTHING.
     * Returns true if this is a new signal (should be processed).
     * Returns false if already seen (skip processing).
     */
    export async function deduplicateSignal(
      provider: string,
      signalId: string,
      workspaceId: string
    ): Promise<boolean> {
      const result = await db
        .insert(processedSignals)
        .values({ provider, signalId, workspaceId })
        .onConflictDoNothing()
        .returning({ id: processedSignals.id });

      return result.length > 0; // true = new signal, false = already processed
    }
    ```
