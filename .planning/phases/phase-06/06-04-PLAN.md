   ```typescript
   /**
    * Cursor-based pagination using composite cursors (createdAt + id).
    * Handles sort column ties per Pitfall 13 from research.
    */

   interface CursorData {
     createdAt: string; // ISO timestamp
     id: string;        // record UUID
   }

   export function encodeCursor(data: CursorData): string {
     return Buffer.from(JSON.stringify(data)).toString("base64url");
   }

   export function decodeCursor(cursor: string): CursorData | null {
     try {
       return JSON.parse(Buffer.from(cursor, "base64url").toString("utf8"));
     } catch {
       return null;
     }
   }
   ```

     ```typescript
     const loadMore = useCallback(async () => {

       setLoadingMore(true);
       try {

           const data = await res.json();
           setRecords(prev => [...prev, ...data.data.records]);
           setNextCursor(data.data.pagination.nextCursor ?? null);
           setHasMore(data.data.pagination.hasMore);
         }
       } finally {
         setLoadingMore(false);
       }
     ```
