export const workspaceAiUsage = pgTable(
  "workspace_ai_usage",
  {
    id: text("id").primaryKey().$defaultFn(() => crypto.randomUUID()),
    workspaceId: text("workspace_id")
      .notNull()
      .references(() => workspaces.id, { onDelete: "cascade" }),
    /** Date string YYYY-MM-DD for daily bucketing */
    date: text("date").notNull(),
    /** Number of generation jobs completed today */
    generationCount: integer("generation_count").notNull().default(0),
    /** Estimated cost in USD cents (integer to avoid float issues) */
    estimatedCostCents: integer("estimated_cost_cents").notNull().default(0),
    updatedAt: timestamp("updated_at").notNull().defaultNow(),
  },
  (table) => [
    uniqueIndex("workspace_ai_usage_ws_date").on(table.workspaceId, table.date),
    index("workspace_ai_usage_workspace_id").on(table.workspaceId),
  ]
);

```typescript
import { db } from "@/db";
import { workspaceAiUsage } from "@/db/schema";
import { and, eq } from "drizzle-orm";

// Default daily budget: 100 generations or $5 (500 cents)
const DEFAULT_DAILY_GENERATION_LIMIT = 100;
const DEFAULT_DAILY_COST_LIMIT_CENTS = 500;

// Estimated cost per generation in cents by context tier
const COST_ESTIMATES: Record<string, number> = {
  light: 2,   // ~$0.02 per light-tier generation (haiku-class)
  full: 10,   // ~$0.10 per full-tier generation (sonnet-class)
};

function todayString(): string {
  return new Date().toISOString().slice(0, 10); // YYYY-MM-DD
}

/**
 * Check if the workspace has budget remaining for today.
 * Returns { allowed: boolean, reason?: string, usage: { count, costCents } }.
 */
export async function checkBudget(
  workspaceId: string
): Promise<{ allowed: boolean; reason?: string; usage: { count: number; costCents: number } }> {
  const date = todayString();

  const [usage] = await db
    .select()
    .from(workspaceAiUsage)
    .where(and(eq(workspaceAiUsage.workspaceId, workspaceId), eq(workspaceAiUsage.date, date)))
    .limit(1);

  const count = usage?.generationCount ?? 0;
  const costCents = usage?.estimatedCostCents ?? 0;

  // TODO: Allow workspace-level override via workspace.settings.aiBudget
  if (count >= DEFAULT_DAILY_GENERATION_LIMIT) {
    return { allowed: false, reason: `Daily generation limit reached (${count}/${DEFAULT_DAILY_GENERATION_LIMIT})`, usage: { count, costCents } };
  }
  if (costCents >= DEFAULT_DAILY_COST_LIMIT_CENTS) {
    return { allowed: false, reason: `Daily cost limit reached ($${(costCents / 100).toFixed(2)}/$${(DEFAULT_DAILY_COST_LIMIT_CENTS / 100).toFixed(2)})`, usage: { count, costCents } };
  }

  return { allowed: true, usage: { count, costCents } };
}

/**
 * Record a generation usage event. Called after successful generation.
 * Uses upsert to atomically increment the daily counter.
 */
export async function recordUsage(
  workspaceId: string,
  contextTier: string
): Promise<void> {
  const date = todayString();
  const costEstimate = COST_ESTIMATES[contextTier] ?? COST_ESTIMATES.light;

  // Upsert: insert if no row for today, otherwise increment
  await db
    .insert(workspaceAiUsage)
    .values({
      workspaceId,
      date,
      generationCount: 1,
      estimatedCostCents: costEstimate,
    })
    .onConflictDoUpdate({
      target: [workspaceAiUsage.workspaceId, workspaceAiUsage.date],
      set: {
        generationCount: sql`${workspaceAiUsage.generationCount} + 1`,
        estimatedCostCents: sql`${workspaceAiUsage.estimatedCostCents} + ${costEstimate}`,
        updatedAt: new Date(),
      },
    });
}

/**
 * Get daily usage for a workspace. Used by settings/dashboard pages.
 */
export async function getDailyUsage(
  workspaceId: string,
  date?: string
): Promise<{ count: number; costCents: number }> {
  const d = date ?? todayString();
  const [usage] = await db
    .select()
    .from(workspaceAiUsage)
    .where(and(eq(workspaceAiUsage.workspaceId, workspaceId), eq(workspaceAiUsage.date, d)))
    .limit(1);

  return { count: usage?.generationCount ?? 0, costCents: usage?.estimatedCostCents ?? 0 };
}
```

Create `apps/web/src/services/documents/dedup-guard.ts`:

This service checks if a generation job for the same (recordId, assetType) was created within the last 15 minutes. If so, it returns `true` (is duplicate, skip generation).

```typescript
import { db } from "@/db";
import { generatedAssets } from "@/db/schema";
import { and, eq, gte } from "drizzle-orm";
import type { AssetType } from "@/db/schema/generated-assets";

const DEDUP_WINDOW_MS = 15 * 60 * 1000; // 15 minutes

/**
 * Check if a generation for the same record + asset type exists within the dedup window.
 * Returns true if a duplicate exists (caller should skip generation).
 *
 * For battlecards, also checks competitorName in metadata to allow different
 * competitors to generate even within the window.
 */
export async function isDuplicate(
  workspaceId: string,
  recordId: string,
  assetType: AssetType,
  competitorName?: string
): Promise<boolean> {
  const windowStart = new Date(Date.now() - DEDUP_WINDOW_MS);

  const recent = await db
    .select({ id: generatedAssets.id, metadata: generatedAssets.metadata })
    .from(generatedAssets)
    .where(
      and(
        eq(generatedAssets.workspaceId, workspaceId),
        eq(generatedAssets.recordId, recordId),
        eq(generatedAssets.assetType, assetType),
        gte(generatedAssets.createdAt, windowStart)
      )
    )
    .limit(5);

  if (recent.length === 0) return false;

  // For battlecards, check if this specific competitor already has a recent card
  if (assetType === "battlecard" && competitorName) {
    return recent.some((r) => {
      const meta = r.metadata as Record<string, unknown> | null;
      return meta?.competitorName === competitorName;
    });
  }

  return true;
}
```

The dedup guard uses the existing `generated_assets_record_type` index on `(recordId, assetType)` for efficient lookups.
