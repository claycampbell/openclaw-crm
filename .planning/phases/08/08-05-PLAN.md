1. Add a `getHotLeads()` function to `apps/web/src/services/activity-scoring.ts` that joins `record_scores` with `records` and key `record_values` (name/title/email) to return enriched hot lead data:

```typescript
/**
 * Get hot leads: top scored records enriched with display name, email, and object type.
 * Returns data ready for the hot leads dashboard.
 */
export async function getHotLeads(
  workspaceId: string,
  limit: number = 20
): Promise<HotLead[]> {
  const topScores = await db
    .select()
    .from(recordScores)
    .innerJoin(records, eq(records.id, recordScores.recordId))
    .where(
      and(
        eq(recordScores.workspaceId, workspaceId),
        // Only include records with score > 0
        sql`${recordScores.score} > 0`
      )
    )
    .orderBy(desc(recordScores.score))
    .limit(limit);

  // Enrich with record display values (name, email)
  const enriched: HotLead[] = [];
  for (const row of topScores) {
    const values = await db
      .select({ slug: attributes.slug, textValue: recordValues.textValue })
      .from(recordValues)
      .innerJoin(attributes, eq(recordValues.attributeId, attributes.id))
      .where(eq(recordValues.recordId, row.record_scores.recordId));

    const nameVal = values.find(v =>
      v.slug.includes("name") || v.slug.includes("full-name") || v.slug.includes("first-name")
    )?.textValue;
    const emailVal = values.find(v => v.slug.includes("email"))?.textValue;
    const companyVal = values.find(v => v.slug.includes("company"))?.textValue;

    const trend = row.record_scores.previousScore !== null
      ? row.record_scores.score - row.record_scores.previousScore
      : 0;

    enriched.push({
      recordId: row.record_scores.recordId,
      score: row.record_scores.score,
      fitScore: row.record_scores.fitScore,
      engagementScore: row.record_scores.engagementScore,
      recencyScore: row.record_scores.recencyScore,
      trend,
      trendDirection: trend > 2 ? "up" : trend < -2 ? "down" : "stable",
      name: nameVal ?? "(unnamed)",
      email: emailVal ?? null,
      company: companyVal ?? null,
      objectSlug: row.records.objectId, // Will need to resolve to slug for linking
      calculatedAt: row.record_scores.calculatedAt,
    });
  }

  return enriched;
}

export interface HotLead {
  recordId: string;
  score: number;
  fitScore: number;
  engagementScore: number;
  recencyScore: number;
  trend: number;
  trendDirection: "up" | "down" | "stable";
  name: string;
  email: string | null;
  company: string | null;
  objectSlug: string;
  calculatedAt: Date;
}
```

NOTE: The `objectSlug` resolution (from objectId to slug for building record links) may need a join through the `objects` table. Use `records.objectId` and join to `objects` to get the slug for link generation. Adjust the query to include this join.

```typescript
  const limit = Math.min(parseInt(searchParams.get("limit") ?? "20"), 50);
```

1. Create `apps/web/src/components/hot-leads/score-badge.tsx` -- a client component:

A small badge showing the score (0-100) with color coding:
- 80-100: green (bg-green-100 text-green-800)
- 50-79: yellow (bg-yellow-100 text-yellow-800)
- 20-49: orange (bg-orange-100 text-orange-800)
- 0-19: gray (bg-gray-100 text-gray-600)

Plus a trend indicator:
- "up": green upward arrow (ArrowUpRight from lucide-react)
- "down": red downward arrow (ArrowDownRight)
- "stable": gray dash (Minus)

```tsx
"use client";

import { ArrowUpRight, ArrowDownRight, Minus } from "lucide-react";
import { cn } from "@/lib/utils";

interface ScoreBadgeProps {
  score: number;
  trend: "up" | "down" | "stable";
  trendValue?: number;
}

export function ScoreBadge({ score, trend, trendValue }: ScoreBadgeProps) {
  const colorClass = score >= 80 ? "bg-green-100 text-green-800"
    : score >= 50 ? "bg-yellow-100 text-yellow-800"
    : score >= 20 ? "bg-orange-100 text-orange-800"
    : "bg-gray-100 text-gray-600";

  const TrendIcon = trend === "up" ? ArrowUpRight : trend === "down" ? ArrowDownRight : Minus;
  const trendColor = trend === "up" ? "text-green-600" : trend === "down" ? "text-red-600" : "text-gray-400";

  return (
    <div className="flex items-center gap-1.5">
      <span className={cn("inline-flex items-center rounded-full px-2.5 py-0.5 text-xs font-semibold", colorClass)}>
        {score}
      </span>
      <span className={cn("flex items-center text-xs", trendColor)}>
        <TrendIcon className="h-3.5 w-3.5" />
        {trendValue !== undefined && trendValue !== 0 && (
          <span className="ml-0.5">{trendValue > 0 ? "+" : ""}{trendValue}</span>
        )}
      </span>
    </div>
  );
}
```

```tsx
"use client";

import { Table, TableBody, TableCell, TableHead, TableHeader, TableRow } from "@/components/ui/table";
import { ScoreBadge } from "./score-badge";
import Link from "next/link";

interface HotLead {
  recordId: string;
  score: number;
  fitScore: number;
  engagementScore: number;
  recencyScore: number;
  trend: number;
  trendDirection: "up" | "down" | "stable";
  name: string;
  email: string | null;
  company: string | null;
  objectSlug: string;
  calculatedAt: string;
}

export function HotLeadsTable({ leads }: { leads: HotLead[] }) {
  if (leads.length === 0) {
    return (
      <div className="text-center py-12 text-muted-foreground">
        <p className="text-lg font-medium">No scored records yet</p>
        <p className="text-sm mt-1">Activity scores are calculated as signals flow in from emails, meetings, and deal activity.</p>
      </div>
    );
  }

  return (
    <Table>
      <TableHeader>
        <TableRow>
          <TableHead className="w-12">#</TableHead>
          <TableHead>Name</TableHead>
          <TableHead>Company</TableHead>
          <TableHead>Email</TableHead>
          <TableHead>Score</TableHead>
          <TableHead className="text-right">Fit / Engage / Recent</TableHead>
        </TableRow>
      </TableHeader>
      <TableBody>
        {leads.map((lead, index) => (
          <TableRow key={lead.recordId}>
            <TableCell className="font-mono text-muted-foreground">{index + 1}</TableCell>
            <TableCell>
              <Link
                href={`/objects/${lead.objectSlug}/${lead.recordId}`}
                className="font-medium text-primary hover:underline"
              >
                {lead.name}
              </Link>
            </TableCell>
            <TableCell className="text-muted-foreground">{lead.company ?? "-"}</TableCell>
            <TableCell className="text-muted-foreground text-sm">{lead.email ?? "-"}</TableCell>
            <TableCell>
              <ScoreBadge score={lead.score} trend={lead.trendDirection} trendValue={lead.trend} />
            </TableCell>
            <TableCell className="text-right text-xs text-muted-foreground">
              {lead.fitScore} / {lead.engagementScore} / {lead.recencyScore}
            </TableCell>
          </TableRow>
        ))}
      </TableBody>
    </Table>
  );
}
```
